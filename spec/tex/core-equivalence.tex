\subsection{Equivalence rules}

\begin{figure}[!htb]
\begin{center}
\framebox{
\begin{minipage}{0.95\columnwidth}
\textbf{Equivalance rules}
\begin{mathpar}

  \inferrule[Union-$\E$-Row-Row]{}{
     \{Op_i | Op_i \in \{Op_1, ..., Op_n\} \lor \\Op_i \in \{Op_{n+1}, ..., Op_{n+m}\}, (\delta_1 \union \delta_2)\} =\\
     \{Op_1, ..., Op_n, \delta_1\} \union \{Op_{n+1}, ..., Op_{n+m}, \delta_2\}
  }

  \inferrule[Union-$\E$-Row-Dot]{}{
     \{Op_i | Op_i \in \{Op_1, ..., Op_n\} \lor \\Op_i \in \{Op_{n+1}, ..., Op_{n+m}\}, \delta_1\} =\\
     \{Op_1, ..., Op_n, \delta_1\} \union \{Op_{n+1}, ..., Op_{n+m}, .\}
  }

  \inferrule[Union-$\E$-Dot-Row]{}{
     \{Op_i | Op_i \in \{Op_1, ..., Op_n\} \lor \\Op_i \in \{Op_{n+1}, ..., Op_{n+m}\}, \delta_2\} =\\
     \{Op_1, ..., Op_n, .\} \union \{Op_{n+1}, ..., Op_{n+m}, \delta_2\}
  }

  \inferrule[Union-$\E$-Dot-Dot]{}{
     \{Op_i | Op_i \in \{Op_1, ..., Op_n\} \lor \\Op_i \in \{Op_{n+1}, ..., Op_{n+m}\}, .\} =\\
     \{Op_1, ..., Op_n, .\} \union \{Op_{n+1}, ..., Op_{n+m}, .\}
  }

  \inferrule[Intersection-$\E$-Row-Row]{}{
    \{Op_i | Op_i \in \{Op_1, ..., Op_n\} \land \\Op_i \in \{Op_{n+1}, ..., Op_{n+m}\}, (\delta_1 \intersection \delta_2)\} =\\
    \{Op_1, ..., Op_n, \delta_1\} \intersection \{Op_{n+1}, ..., Op_{n+m}, \delta_2\}
  }

  \inferrule[Intersection-$\E$-Row-Dot]{}{
    \{Op_i | Op_i \in \{Op_1, ..., Op_n\} \land \\Op_i \in \{Op_{n+1}, ..., Op_{n+m}\}, \delta_1\} =\\
    \{Op_1, ..., Op_n, \delta_1\} \intersection \{Op_{n+1}, ..., Op_{n+m}, .\}
  }

  \inferrule[Intersection-$\E$-Dot-Row]{}{
    \{Op_i | Op_i \in \{Op_1, ..., Op_n\} \land \\Op_i \in \{Op_{n+1}, ..., Op_{n+m}\}, \delta_2\} =\\
    \{Op_1, ..., Op_n, .\} \intersection \{Op_{n+1}, ..., Op_{n+m}, \delta_2\}
  }

  \inferrule[Intersection-$\E$-Dot-Dot]{}{
    \{Op_i | Op_i \in \{Op_1, ..., Op_n\} \land \\Op_i \in \{Op_{n+1}, ..., Op_{n+m}\}, .\} =\\
    \{Op_1, ..., Op_n, .\} \intersection \{Op_{n+1}, ..., Op_{n+m}, .\}
  }

\end{mathpar}
\end{minipage}
}
